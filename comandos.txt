# Construir a imagem docker
sudo docker build -t teste-devops .

# Executar o container
sudo docker run -d -p 8080:80 teste-devops

# Acessar a seguinte URL no navegador:
http://localhost:8080

# Listar containers em execução
sudo docker ps

# Parar container
sudo docker stop <container_id>

# Remover container
sudo docker rm <container_id>

# Remover imagem
sudo docker rmi teste-devops
